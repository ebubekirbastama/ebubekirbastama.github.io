window.onload = function () { var watchID = null; var marker = L.marker(); var path = []; var isTracking = false; var map = L.map('map').fitWorld(); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { maxZoom: 19, attribution: '© OpenStreetMap contributors' }).addTo(map); // Kabe'nin konumu var kabe = L.marker([21.4225, 39.8262]).addTo(map); kabe.bindPopup("

Kabe

"); // Cihazın konum takibi function startTracking() { isTracking = true; navigator.geolocation.getCurrentPosition(function (location) { marker.setLatLng([location.coords.latitude, location.coords.longitude]).addTo(map); map.setView([location.coords.latitude, location.coords.longitude], 18); path.push(marker.getLatLng()); path.push(kabe.getLatLng()); L.polyline(path, {color: 'green'}).addTo(map); }); if (watchID == null) { watchID = navigator.geolocation.watchPosition(function (location) { if (isTracking) { marker.setLatLng([location.coords.latitude, location.coords.longitude]); // Cihazın yönü hesaplanıyor ve marker'ın yönü güncelleniyor if (window.DeviceOrientationEvent) { window.addEventListener("deviceorientation", handleOrientation, true); } else { console.log("Device orientation not supported."); } path.push(marker.getLatLng()); L.polyline(path, {color: 'red'}).addTo(map); } }); } } function stopTracking() { isTracking = false; } function handleOrientation(event) { var alpha = event.alpha; // Yaw (0-360) marker.setRotationAngle(alpha); } startTracking(); }
